def get_cf_worker_domains() -> list[dict[str, Any]]:
    raw = get_setting("cf_worker_domains", "[]")
    try:
        return json.loads(raw)
    except Exception:
        return []

def get_cf_worker_default_domain() -> str:

let currentSettingsTab = 'basic';

function switchSettingsTab(tabName) {
    const prevTab = currentSettingsTab;
    if (prevTab === tabName) return; // 点击同一 Tab 无操作

    // 1. 立即切换 Tab UI（不等待保存完成）

        // ...
    }

    let dotEl = null;

    if (prevTabBtn) {
        // 添加黄色保存中圆点
        dotEl = document.createElement('span');
        dotEl.className = 'tab-save-dot saving-dot';
        prevTabBtn.appendChild(dotEl);
        prevTabBtn.classList.add('saving');
        prevTabBtn.classList.remove('save-error');
    }

    try {
        const resp = await fetch('/api/settings', {
            method: 'PUT',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(settings)
        });
        if (!resp.ok) throw new Error(`HTTP ${resp.status}`);

        // 保存成功：移除圆点 + saving class
        if (prevTabBtn) {
            prevTabBtn.classList.remove('saving');
            if (dotEl) dotEl.remove();

## 关键修正点（经验教训）

1. **同步 API 路由**：`/api/settings/cf-worker-sync-domains`（已确认 `routes/settings.py`），不是 `/sync-cf-worker-domains`
2. **字段 ID 对齐**：API 安全 Tab 和自动化 Tab 的所有字段 ID 与现有代码完全一致，**不重命名**
3. **只读字段永远只读**：CF Worker 域名字段从初始化到同步后始终只读，空值时通过 placeholder 引导
4. **GPTMail API Key ID 更新**：从 `settingsApiKey` 改为 `settingsTempMailApiKey`，且现有脱敏逻辑（`dataset.maskedValue`/`dataset.isSet`）需一并迁移
5. **基础 Tab 不触发自动保存**：密码字段必须在基础 Tab 内手动保存
6. **Tab 切换时序**：立即切换 Tab，后台异步保存（不阻塞 UI），圆点为后台进度指示器
7. **CSS 变量名**：项目使用 `var(--clr-primary)`，**不是** `var(--primary)`（Phase 3 CSS 代码已修正）
8. **syncCfWorkerDomains 元素 ID 变更**：旧 `cfWorkerSyncDomainsHint` → 新 `cfWorkerSyncTime`，提示内容格式也需更新
9. **Tab 视觉结构（UI 定稿）**：Tab 导航栏 + 全部内容区用 `.settings-card-shell` 包裹为一个卡片（"文件夹标签页"隐喻），`.settings-tab-nav` 必须设 `overflow-y: hidden` 防止出现垂直滚动条
10. **保存圆点颜色语义（UI 定稿）**：保存中 = **黄色**（`var(--clr-warn)`），保存失败 = **红色**（`var(--clr-danger)`）；两个状态通过 `.saving` / `.save-error` class 控制
